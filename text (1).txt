    <script>
      function calculateRevenue() {
        const prospect = document.getElementById("prospect").value;
        const prospectLeads = document.getElementById("prospectLeads").value / 100;
        const conversion = document.getElementById("conversion").value / 100;
        const deal = document.getElementById("deal").value;

        const leads = prospect * prospectLeads;
        const opportunities = leads * conversion;
        const demand = opportunities * deal;

        const result = document.getElementById("result");
        result.innerHTML = "<p>Leads Generated: " + leads.toFixed(2) + "</p>" +
                           "<p>Opportunities Created: " + opportunities.toFixed(2) + "</p>" +
                           "<p>Demand Generated: " + demand.toFixed(2) + "</p>";
      }
    </script>
